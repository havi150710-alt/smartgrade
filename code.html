<script>

function calculate() {

    const ids = ["toan","van","anh","ly","hoa","sinh"];

    let values = [];

    for (let id of ids) {

        let value = document.getElementById(id).value;

        if (value === "") {

            alert("Vui lòng nhập đầy đủ tất cả các môn!");

            return;

        }

        if (value < 0 || value > 10) {

            alert("Điểm phải nằm trong khoảng 0 đến 10.");

            return;

        }

        values.push(parseFloat(value));

    }

    let avg = (values[0]*2 + values[1]*2 + values[2] + values[3] + values[4] + values[5]) / 8;

    avg = avg.toFixed(2);

    let rank, color, advice;

    if (avg >= 8) {

        rank = "Giỏi";

        color = "#28a745";

        advice = "Xuất sắc! Hãy tiếp tục duy trì phong độ này.";

    }

    else if (avg >= 6.5) {

        rank = "Khá";

        color = "#007BFF";

        advice = "Bạn đang làm tốt, hãy cải thiện thêm các môn còn yếu.";

    }

    else if (avg >= 5) {

        rank = "Trung bình";

        color = "#fd7e14";

        advice = "Nên dành thêm thời gian ôn tập để nâng cao kết quả.";

    }

    else {

        rank = "Yếu";

        color = "#dc3545";

        advice = "Cần lập kế hoạch học tập rõ ràng và nghiêm túc hơn.";

    }

    const resultBox = document.getElementById("resultBox");

    resultBox.style.display = "block";

    resultBox.style.background = color + "15";

    document.getElementById("average").innerHTML = "<b>Điểm trung bình:</b> " + avg;

    document.getElementById("rank").innerText = "Xếp loại: " + rank;

    document.getElementById("rank").style.color = color;

    document.getElementById("advice").innerText = advice;

    drawChart(avg);

}

function drawChart(avg) {

    const canvas = document.getElementById("chart");

    const ctx = canvas.getContext("2d");

    ctx.clearRect(0,0,200,200);

    const percent = avg * 10;

    const angle = (percent/100) * 2 * Math.PI;

    ctx.beginPath();

    ctx.moveTo(100,100);

    ctx.arc(100,100,80,0,angle);

    ctx.fillStyle = "#007BFF";

    ctx.fill();

    ctx.beginPath();

    ctx.moveTo(100,100);

    ctx.arc(100,100,80,angle,2*Math.PI);

    ctx.fillStyle = "#e9ecef";

    ctx.fill();

    ctx.fillStyle = "#333";

    ctx.font = "20px Arial";

    ctx.textAlign = "center";

    ctx.fillText(avg, 100, 105);

}
